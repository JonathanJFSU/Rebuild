FROM ubuntu:20.04

# Set timezone and disable prompts
ENV TZ=America/New_York \
    DEBIAN_FRONTEND=noninteractive

# Use bash as default shell
SHELL ["/bin/bash", "--login", "-c"]

# Install OS dependencies
RUN apt-get update && apt-get install -y \
    wget \
    libgl1 \
    ca-certificates \
    && rm -rf /var/lib/apt/lists/*

# Set working directory
WORKDIR /app

# Copy files into container
COPY environment.yml .
COPY paul.py .
COPY testVideo.mp4 .
COPY matrix.py .

# Install Miniconda (Python + Conda base)
RUN wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh && \
    bash Miniconda3-latest-Linux-x86_64.sh -b -p /opt/conda && \
    rm Miniconda3-latest-Linux-x86_64.sh

# Add conda to path
ENV PATH=/opt/conda/bin:$PATH

# Create the conda environment from .yml
RUN conda env create -f environment.yml

# Activate environment for all future shells
ENV CONDA_DEFAULT_ENV=detections
ENV PATH=/opt/conda/envs/detections/bin:$PATH

# Ensure conda activates on container start
RUN echo "source activate detections" >> ~/.bashrc

# Set default entrypoint
ENTRYPOINT ["/bin/bash"]
